#Part B
#Display a line graph showing the change in average city vs. highway MPG over time. 
#Do not round the data to integers this time.  Show the city MPG as a red line 
#and the highway MPG as a blue line.
#You do not need to add a legend to this diagram.  The structure of the data 
#would make this somewhat complicated.  Well discuss that more when we get to 
#data wrangling.

vehicles %>%
  group_by(year) %>%
  summarise(city_average= mean(citympg), highway_average=mean(highwaympg))%>%
  pivot_longer(cols= !year, names_to = 'Driving_Location', 
               values_to = 'AverageMPG') %>%
  ggplot()+
  geom_line(aes(year, AverageMPG, color=Driving_Location))+
  scale_color_manual(values=c('red', 'blue'))+
  ggtitle("Average MPG per year", subtitle = "All vehicle classes: City vs Highway")

#Part C
#Modify the graph above to also show the overall MPG, computed as the average of
#city and highway MPG.  Plot this as a green line.

vehicles %>%
  group_by(year) %>%
  summarise(city_average= mean(citympg), highway_average=mean(highwaympg),
            combined_average=mean(city_average + highway_average)/2) %>%
  pivot_longer(cols= !year, names_to = 'Driving_Location', 
               values_to = 'AverageMPG') %>%
  ggplot()+
  geom_line(aes(year, AverageMPG, color=Driving_Location))+
  scale_color_manual(values=c('red', 'green', 'blue'))+
  ggtitle("Average MPG per year", subtitle = "All vehicle classes: City vs Highway")+
  theme_light()

#Part D
#Modify the graph above to show separate graphs for each drive type.

vehicles %>%
  group_by(year, drive) %>%
  summarise(city_average= mean(citympg), highway_average=mean(highwaympg),
            combined_average=mean(city_average + highway_average)/2) %>%
  pivot_longer(cols= !c(year, drive), names_to = 'Driving_Location', 
               values_to = 'AverageMPG') %>%
  ggplot()+
  geom_line(aes(year, AverageMPG, color=Driving_Location))+
  scale_color_manual(values= c('red', 'green', 'blue'))+
  facet_wrap(~drive)+
  ggtitle("Average MPG per year by drive type", subtitle = "All vehicle classes: City vs Highway")+
  theme_light()
